import streamlit as st import random import smtplib from email.message import EmailMessage # --- EMAIL CONFIGURATION (The Secret Step) --- # Replace these with your actual details later EMAIL_SENDER = "your-email@gmail.com" EMAIL_PASSWORD = "your-app-password" def send_otp(target_email, code): msg = EmailMessage() msg.set_content(f"Your Bagong Pilipinas verification code is: {code}") msg['Subject'] = 'Verification Code' msg['From'] = EMAIL_SENDER msg['To'] = target_email try: server = smtplib.SMTP_SSL('smtp.gmail.com', 465) server.login(EMAIL_SENDER, EMAIL_PASSWORD) server.send_message(msg) server.quit() return True except Exception as e: st.error(f"Email error: {e}") return False # --- APP UI --- st.set_page_config(page_title="Bagong Pilipinas Stock Market", page_icon="🇵🇭") if 'step' not in st.session_state: st.session_state.step = "register" # STEP 1: REGISTRATION if st.session_state.step == "register": st.title("🇵🇭 Register Account") email = st.text_input("Enter your Email Address") if st.button("SEND 6-DIGIT CODE"): if "@" in email: otp = str(random.randint(100000, 999999)) st.session_state.generated_otp = otp st.session_state.user_email = email if send_otp(email, otp): st.success(f"Code sent to {email}") st.session_state.step = "verify" st.rerun() else: st.error("Please enter a valid email.") # STEP 2: OTP VERIFICATION elif st.session_state.step == "verify": st.title("Enter Code") st.write(f"Check your email: **{st.session_state.user_email}**") user_code = st.text_input("6-Digit Code", max_chars=6) if st.button("VERIFY & ENTER"): if user_code == st.session_state.generated_otp: st.success("Verified!") st.session_state.step = "dashboard" st.rerun() else: st.error("Invalid Code. Try again.") # STEP 3: THE ACTUAL APP elif st.session_state.step == "dashboard": st.title("📈 Bagong Pilipinas Dashboard") st.write(f"Welcome, {st.session_state.user_email}!") if st.button("Log Out"): st.session_state.step = "register" st.rerun()
